\begin{algorithm}[H]
\DontPrintSemicolon
\SetAlgoLined
\SetKwInOut{Input}{Input}\SetKwInOut{Output}{Output}
\Input{ 
\begin{itemize}[label={--}]
\item set of compliance rule $crList$
\item set of parametric constraints $params$
\end{itemize}
}
\SetKwProg{Fn}{Function}{}{}
\SetKw{Continue}{continue}
\Begin{
	initialize ComplianceController\;
	initialize ChoreographyModelGenerator\;
	initialize CollaborationGenerator\;
	buildSuccess = \textbf{false}\;
	\While{!buildSuccess}{
		$model$ = choreographyModelGenerator.build($params$)\;
		\eIf{$crList$ \ne empty}{
			$buildSucess$ = complianceController.assign()\;
		}{
			$buildSuccess$ = $true$\;
		}
		\If{buildSuccess \ne true \land iterationCount \bmod 10 \equiv 0}{
			$params$.interactionCount = $params$.interactionCount * 1.1\;
		}
	}
	choreographyGenerator.generateChoreographyFromModel($model$)\;
}
\end{algorithm}
