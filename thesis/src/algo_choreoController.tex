\Begin{
	initialize SplitTracking\;
	initialize ComplianceController\;
	initialize ChoreographyModelGenerator\;
	initialize CollaborationGenerator\;
	buildSuccess = FALSE;
	\While{!buildSucess}{
		model = choreographyModelGenerator.build()\;
		\eIf{compliance rules are defined}{
			buildSucess = complianceController.assign()\;
		}{
			buildSuccess = TRUE\;
		}

			interactionCount = interactionCount * 1.1\;
		}
	}
	choreographyGenerator.generateChoreographyFromModel(model)\;
}
